## godoc authoring conventions (apply in every Part A task)

1. Every exported symbol gets a doc comment starting with the symbol name
   (`// GetFrontierMomentum returns ...`). Package comment goes in the file
   that names the package's purpose best (`ledger.go` for `api`,
   `shared.go` for `embedded`, `api.go` for `subscribe`) — no new files;
   a docs PR adds zero files.
2. Comments document behavior derivable from the code: parameter meaning
   and units (heights are 1-based momentum heights; `count` is a maximum;
   amounts are raw base-unit `*big.Int` serialized as strings), pagination
   semantics (`pageIndex` is 0-based; note the max `pageSize` constant the
   handler enforces — find it in the function body, e.g.
   `api.checkTokenPageParams`-style guards), sort order (**read the
   comparison operator yourself**: `<` on height = ascending — the old v1
   docs got sort directions wrong twice), nil-vs-error behavior ("returns
   nil and no error when no block exists" — verify per method), and RPC
   method names (each Go method maps to JSON-RPC `<namespace>.<lowerCamel>`,
   e.g. `LedgerApi.GetFrontierMomentum` → `ledger.getFrontierMomentum`).
3. Never restate the signature ("GetByName gets by name" is banned).
   If a method's body just delegates, say what the caller observably gets.
4. Comment-only discipline: no identifier, import, signature, or
   whitespace-inside-code changes. The checker (Task A1) enforces this.
5. Cold authoring: do not consult the deleted v1 text (`refs/pull/2/head`).

```bash
#!/usr/bin/env bash
# check-comment-only.sh BASE_REF
# Verifies every .go file changed since BASE_REF differs only in comments:
# both versions are parsed, comments dropped, pretty-printed, and compared.
set -euo pipefail
BASE="${1:?usage: check-comment-only.sh BASE_REF}"
cd "$(git rev-parse --show-toplevel)"

TMP="$(mktemp -d)"
trap 'rm -rf "$TMP"' EXIT
cat > "$TMP/strip.go" <<'EOF'
package main

import (
	"fmt"
	"go/parser"
	"go/printer"
	"go/token"
	"os"
)

// Reads a Go file, drops all comments, prints canonical source to stdout.
func main() {
	src, err := os.ReadFile(os.Args[1])
	if err != nil {
		fmt.Fprintln(os.Stderr, err)
		os.Exit(1)
	}
	fset := token.NewFileSet()
	f, err := parser.ParseFile(fset, os.Args[1], src, 0) // no ParseComments
	if err != nil {
		fmt.Fprintln(os.Stderr, err)
		os.Exit(1)
	}
	if err := (&printer.Config{Mode: printer.TabIndent, Tabwidth: 8}).Fprint(os.Stdout, fset, f); err != nil {
		fmt.Fprintln(os.Stderr, err)
		os.Exit(1)
	}
}
EOF

fail=0
while IFS= read -r f; do
	[[ "$f" == *.go ]] || continue
	git show "$BASE:$f" > "$TMP/old.go" 2>/dev/null || { echo "NEW FILE (not allowed in docs PR): $f"; fail=1; continue; }
	GOWORK=off go run "$TMP/strip.go" "$TMP/old.go" > "$TMP/old.stripped" || { echo "PARSE FAIL (base): $f"; fail=1; continue; }
	GOWORK=off go run "$TMP/strip.go" "$f"          > "$TMP/new.stripped" || { echo "PARSE FAIL (head): $f"; fail=1; continue; }
	if ! diff -q "$TMP/old.stripped" "$TMP/new.stripped" > /dev/null; then
		echo "NON-COMMENT CHANGE: $f"
		diff "$TMP/old.stripped" "$TMP/new.stripped" | head -10
		fail=1
	fi
done < <(git diff --name-only "$BASE" -- '*.go')

[[ $fail -eq 0 ]] && echo "OK: all changed .go files are comment-only vs $BASE"
exit $fail
```

```bash
# Clones the pinned go-zenon ref into .build/go-zenon. Idempotent.
set -euo pipefail

source <(sed 's/^/PIN_/' PINNED_GO_ZENON)   # PIN_repo, PIN_ref
DEST=.build/go-zenon

if [[ -d "$DEST/.git" ]] && [[ "$(git -C "$DEST" rev-parse HEAD)" == "$PIN_ref" ]]; then
	echo "fetch-go-zenon: already at $PIN_ref"

fi
rm -rf "$DEST" && mkdir -p "$DEST"
git -C "$DEST" init -q
git -C "$DEST" fetch -q --depth 1 "$PIN_repo" "$PIN_ref"
git -C "$DEST" checkout -q FETCH_HEAD
[[ "$(git -C "$DEST" rev-parse HEAD)" == "$PIN_ref" ]] || { echo "fetch-go-zenon: HEAD != pinned ref" >&2; exit 1; }
echo "fetch-go-zenon: checked out $PIN_ref"
```

```bash
#!/usr/bin/env bash
# Generates docs/reference/packages/<pkg>.md from the pinned go-zenon
# checkout using gomarkdoc. Only packages listed in PACKAGES are
# generated (expand as godoc layers land in go-zenon).
set -euo pipefail
cd "$(dirname "$0")/.."
source <(sed 's/^/PIN_/' PINNED_GO_ZENON)
GOMARKDOC_VERSION=v1.1.0
PACKAGES=(rpc/api rpc/api/embedded rpc/api/subscribe)

GOBIN="$PWD/.build/bin" go install "github.com/princjef/gomarkdoc/cmd/gomarkdoc@${GOMARKDOC_VERSION}"
OUT="$PWD/docs/reference/packages"
rm -rf "$OUT" && mkdir -p "$OUT"

pos=1
for pkg in "${PACKAGES[@]}"; do
	slug="${pkg//\//-}"
	(cd .build/go-zenon && GOWORK=off "$PWD/../bin/gomarkdoc" \
		--output "$OUT/$slug.md" \
		--repository.url "https://github.com/0x3639/go-zenon" \
		--repository.default-branch "$PIN_ref" \
		--repository.path "/" \
		"./$pkg")
	# Docusaurus front matter; format: md forces CommonMark parsing so
	# godoc text containing { or < cannot break the MDX compiler.
	tmp="$(mktemp)"
	cat > "$tmp" <<EOF
---
title: "${pkg}"
slug: "/reference/packages/${slug}"
sidebar_label: "${pkg}"
sidebar_position: ${pos}
format: md
---

EOF
	cat "$OUT/$slug.md" >> "$tmp" && mv "$tmp" "$OUT/$slug.md"
	pos=$((pos+1))
done
echo "gen-godoc: generated ${#PACKAGES[@]} package pages at $PIN_ref"
```

```js
// Merges specs/openrpc/*.yaml into specs/openrpc.json, validates it,
// and copies it to static/openrpc.json for publishing.
import { readFileSync, writeFileSync, readdirSync, copyFileSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { parse } from 'yaml';
import { validateOpenRPCDocument } from '@open-rpc/schema-utils-js';

const root = join(dirname(fileURLToPath(import.meta.url)), '..');
const specDir = join(root, 'specs', 'openrpc');
const pin = Object.fromEntries(
  readFileSync(join(root, 'PINNED_GO_ZENON'), 'utf8')
    .trim().split('\n').map(l => l.split('=')),
);

const doc = {
  openrpc: '1.2.6',
  info: {
    title: 'Zenon Network JSON-RPC API',
    version: pin.ref,
    description: 'JSON-RPC 2.0 API of go-zenon, generated from hand-authored fragments verified against the pinned source.',
  },
  methods: [],
  components: { schemas: {} },
};

for (const f of readdirSync(specDir).filter(f => f.endsWith('.yaml')).sort()) {
  const frag = parse(readFileSync(join(specDir, f), 'utf8'));
  if (frag.methods) doc.methods.push(...frag.methods);
  if (frag.schemas) Object.assign(doc.components.schemas, frag.schemas);
}
const compDir = join(specDir, 'components');
for (const f of readdirSync(compDir).filter(f => f.endsWith('.yaml'))) {
  Object.assign(doc.components.schemas, parse(readFileSync(join(compDir, f), 'utf8')).schemas);
}

const names = new Set();
for (const m of doc.methods) {
  if (names.has(m.name)) throw new Error(`duplicate method: ${m.name}`);
  names.add(m.name);
}

const result = validateOpenRPCDocument(doc);
if (result !== true) throw result;

writeFileSync(join(root, 'specs', 'openrpc.json'), JSON.stringify(doc, null, 2) + '\n');
copyFileSync(join(root, 'specs', 'openrpc.json'), join(root, 'static', 'openrpc.json'));
console.log(`assemble-openrpc: ${doc.methods.length} methods, valid OpenRPC ${doc.openrpc}`);
```
